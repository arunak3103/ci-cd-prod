pipeline {
    agent any

    environment {
        AWS_REGION = 'us-east-1'
        ECR_REPO = '112864514073.dkr.ecr.us-east-1.amazonaws.com/web-test'
        IMAGE_NAME = 'ecs-demo'
    }

    stages {

        stage('Checkout') {
            steps {
                git branch: 'main',
                    url: 'https://github.com/arunak3103/ci-cd-prod.git'
            }
        }

        stage('Build Docker Image') {
            steps {
                sh '''
                # Get the short commit hash to tag the image
                COMMIT_HASH=$(git rev-parse --short HEAD)

                # Build the image with both 'latest' and commit hash tags
                docker build -t $IMAGE_NAME:latest -t $IMAGE_NAME:$COMMIT_HASH .
                '''
            }
        }

        stage('Push to ECR') {
            steps {
                sh '''
                # Login to AWS ECR
                aws ecr get-login-password --region $AWS_REGION \
                | docker login --username AWS --password-stdin $ECR_REPO

                # Tag images with ECR repo
                COMMIT_HASH=$(git rev-parse --short HEAD)
                docker tag $IMAGE_NAME:latest $ECR_REPO:latest
                docker tag $IMAGE_NAME:$COMMIT_HASH $ECR_REPO:$COMMIT_HASH

                # Push both tags
                docker push $ECR_REPO:latest
                docker push $ECR_REPO:$COMMIT_HASH
                '''
            }
        }

        stage('Deploy to ECS') {
            steps {
                sh '''
                # Force ECS service update to deploy new image
                aws ecs update-service \
                    --cluster ecs-demo-cluster \
                    --service tomcat-task-service-7mj6347v \
                    --force-new-deployment \
                    --region $AWS_REGION
                '''
            }
        }
    }

    post {
        success {
            echo "✅ Pipeline succeeded"
        }
        failure {
            echo "❌ Pipeline failed"
        }
    }
}
